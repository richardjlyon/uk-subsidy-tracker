```python
    path = ro_derived_dir / f"{grain}.parquet"

        model(**{k: v for k, v in row.to_dict().items() if k in model.model_fields})

    # Column-order discipline: Parquet columns MUST equal model field declaration order (D-10)
    expected_cols = list(model.model_fields.keys())
```

```python
# ============================================================================
# RO byte-identity parametrisation (Plan 05-10; TEST-05; D-21 Phase 4 contract)
# ============================================================================

RO_GRAINS = (
    "station_month",
    "annual_summary",
    "by_technology",
    "by_allocation_round",
    "forward_projection",
)


@pytest.fixture(scope="module")
def ro_derived_once(tmp_path_factory) -> Path:
    from uk_subsidy_tracker.schemes import ro
    out = tmp_path_factory.mktemp("ro-derived-run-1")
    ro.rebuild_derived(output_dir=out)
    return out


@pytest.fixture(scope="module")
def ro_derived_twice(tmp_path_factory) -> Path:
    from uk_subsidy_tracker.schemes import ro
    out = tmp_path_factory.mktemp("ro-derived-run-2")
    ro.rebuild_derived(output_dir=out)
    return out


@pytest.mark.parametrize("grain", RO_GRAINS)
def test_ro_parquet_content_identical(grain, ro_derived_once, ro_derived_twice):
    """TEST-05 / D-21: two consecutive ro.rebuild_derived() calls produce byte-identical Parquet."""
    import pyarrow.parquet as pq
    t1 = pq.read_table(ro_derived_once / f"{grain}.parquet")
    t2 = pq.read_table(ro_derived_twice / f"{grain}.parquet")
    assert t1.equals(t2), f"RO {grain}.parquet content-drift across rebuilds — check deterministic anchors"


@pytest.mark.parametrize("grain", RO_GRAINS)
def test_ro_parquet_writer_metadata(grain, ro_derived_once):
    """TEST-05: Parquet 'created_by' metadata identifies pyarrow writer."""
    import pyarrow.parquet as pq
    meta = pq.read_metadata(ro_derived_once / f"{grain}.parquet")
    created_by = meta.metadata.get(b"pyarrow") or meta.created_by
    assert "parquet" in str(created_by).lower() or "arrow" in str(created_by).lower()
```
