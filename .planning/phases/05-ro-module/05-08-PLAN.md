```python
"""RO dynamics — S2 flagship 4-panel chart.

Panels (CY x-axis per D-07):
  [1] Annual ROC-eligible generation (TWh) — GB-only per D-12
  [2] ROC price (£/ROC) — buyout + recycle band with e-ROC sensitivity overlay (D-02)
  [3] Premium per MWh (£/MWh) — RO cost - gas counterfactual
  [4] Cumulative RO bill (£bn) — with visible mutualisation spike OY 2021-22 (D-11)

Filename: subsidy_ro_dynamics (per RO-MODULE-SPEC Appendix A).
Reads from data/derived/ro/station_month.parquet (NOT raw).
"""
from __future__ import annotations
import pandas as pd
import plotly.graph_objects as go
import pyarrow.parquet as pq

from uk_subsidy_tracker.plotting import ChartBuilder
from uk_subsidy_tracker.schemes import ro


def _prepare() -> pd.DataFrame:
    sm = pq.read_table(ro.DERIVED_DIR / "station_month.parquet").to_pandas()
    gb = sm[sm["country"] == "GB"].copy()
    gb["cy"] = gb["month_end"].dt.year.astype("int64")
    annual = gb.groupby("cy", sort=True, observed=True).agg(
        generation_twh=("generation_mwh", lambda s: s.sum() / 1e6),
        cost_gbp=("ro_cost_gbp", "sum"),
        cost_eroc_gbp=("ro_cost_gbp_eroc", "sum"),
        counterfactual_gbp=("gas_counterfactual_gbp", "sum"),
        premium_gbp=("premium_gbp", "sum"),
        mutualisation_gbp=("mutualisation_gbp", "sum"),
    ).reset_index()
    annual["cumulative_bn"] = (annual["cost_gbp"].cumsum() / 1e9)
    annual["premium_per_mwh"] = annual["premium_gbp"] / (annual["generation_twh"] * 1e6)
    return annual


def main() -> None:
    df = _prepare()
    total_bn = float(df["cumulative_bn"].iloc[-1]) if len(df) else 0.0

    builder = ChartBuilder(
        title="The RO mechanism — ROCs × ROC-price = consumer bill",
        height=1100,
    )
    fig = builder.create_subplots(
        rows=4, cols=1, shared_xaxes=True,
        subplot_titles=[
            "[1] Annual ROC-eligible generation (TWh, GB-only)",
            "[2] ROC price (£/ROC) — buyout+recycle vs e-ROC sensitivity",
            "[3] Premium per MWh (£/MWh) — consumer overpayment vs gas counterfactual",
            f"[4] Cumulative RO bill (£bn) — £{total_bn:.1f}bn to date; mutualisation spike visible OY 2021-22",
        ],
        vertical_spacing=0.06,
    )

    # Panel 1: generation bars
    fig.add_trace(go.Bar(x=df["cy"], y=df["generation_twh"], name="Generation TWh"), row=1, col=1)
    # Panel 2: ROC price bar + e-ROC overlay (sensitivity band)
    price_buyout_recycle = df["cost_gbp"] / ((df["generation_twh"] * 1e6) * 1.5)  # rough £/ROC proxy
    fig.add_trace(go.Scatter(x=df["cy"], y=price_buyout_recycle, mode="lines+markers", name="buyout + recycle"), row=2, col=1)
    # Panel 3: premium per MWh
    fig.add_trace(go.Scatter(x=df["cy"], y=df["premium_per_mwh"], name="Premium £/MWh"), row=3, col=1)
    # Panel 4: cumulative + mutualisation spike annotation
    fig.add_trace(go.Scatter(x=df["cy"], y=df["cumulative_bn"], mode="lines+markers", fill="tozeroy", name="Cumulative £bn"), row=4, col=1)
    # Mutualisation spike annotation on 2021-22 only (D-11)
    mut_year_idx = df.index[df["mutualisation_gbp"].fillna(0) > 0].tolist()
    for idx in mut_year_idx:
        year = int(df.loc[idx, "cy"])
        fig.add_annotation(
            x=year, y=float(df.loc[idx, "cumulative_bn"]),
            text=f"£{df.loc[idx, 'mutualisation_gbp'] / 1e6:.0f}M mutualisation", showarrow=True, row=4, col=1,
        )

    builder.save(fig, "subsidy_ro_dynamics", export_twitter=True)


if __name__ == "__main__":
    main()
```

```python
"""RO cost by technology — S3 stacked-area 2-panel.

Panels (CY x-axis):
  [1] Annual stacked bars by category (£bn per year)
  [2] Cumulative stacked area by category

Categories (D-10 cofiring-vs-dedicated split):
  Onshore Wind, Offshore Wind, Biomass (cofiring), Biomass (dedicated), Solar PV, Other

Filename: subsidy_ro_by_technology.
"""
from __future__ import annotations
import pandas as pd
import plotly.graph_objects as go
import pyarrow.parquet as pq

from uk_subsidy_tracker.plotting import ChartBuilder
from uk_subsidy_tracker.schemes import ro


CATEGORIES = {
    "Onshore Wind": "#6baed6",
    "Offshore Wind": "#1f77b4",
    "Biomass (cofiring)": "#d62728",
    "Biomass (dedicated)": "#8c564b",
    "Solar PV": "#ff7f0e",
    "Other": "#2ca02c",
}

_TECH_MAP = {
    "Onshore wind": "Onshore Wind",
    "Offshore wind": "Offshore Wind",
    "Co-firing biomass": "Biomass (cofiring)",
    "Co-firing biomass (CHP)": "Biomass (cofiring)",
    "Co-firing energy crops": "Biomass (cofiring)",
    "Dedicated biomass": "Biomass (dedicated)",
    "Biomass Conversion": "Biomass (dedicated)",
    "Dedicated energy crops": "Biomass (dedicated)",
    "Solar PV — ground mounted": "Solar PV",
    "Solar PV — building mounted": "Solar PV",
    "Solar photovoltaic (pre-2013)": "Solar PV",
}


def _categorise(tech: str) -> str:
    return _TECH_MAP.get(tech, "Other")


def _prepare() -> pd.DataFrame:
    by_tech = pq.read_table(ro.DERIVED_DIR / "by_technology.parquet").to_pandas()
    by_tech["category"] = by_tech["technology"].map(_categorise)
    return (by_tech.groupby(["year", "category"], sort=True, observed=True)
            ["ro_cost_gbp"].sum().reset_index())


def main() -> None:
    df = _prepare()
    builder = ChartBuilder(title="RO cost by technology (£bn per year, GB)", height=1000)
    fig = builder.create_subplots(rows=2, cols=1, shared_xaxes=True,
                                   subplot_titles=["[1] Annual £bn by category",
                                                   "[2] Cumulative stacked area"])
    # Panel 1 stacked bars
    for cat, color in CATEGORIES.items():
        sub = df[df["category"] == cat]
        fig.add_trace(go.Bar(x=sub["year"], y=sub["ro_cost_gbp"] / 1e9,
                             name=cat, marker_color=color), row=1, col=1)
    fig.update_layout(barmode="stack")
    # Panel 2 cumulative stackgroup
    for cat, color in CATEGORIES.items():
        sub = df[df["category"] == cat].sort_values("year")
        sub = sub.copy()
        sub["cum_bn"] = sub["ro_cost_gbp"].cumsum() / 1e9
        fig.add_trace(go.Scatter(x=sub["year"], y=sub["cum_bn"],
                                  mode="lines", stackgroup="one", line=dict(color=color),
                                  name=cat, showlegend=False), row=2, col=1)

    builder.save(fig, "subsidy_ro_by_technology", export_twitter=True)


if __name__ == "__main__":
    main()
```

```python
"""RO concentration — S4 Lorenz curve on station-level cumulative ROC costs (GB-only per D-12)."""
from __future__ import annotations
import numpy as np
import pandas as pd
import plotly.graph_objects as go
import pyarrow.parquet as pq

from uk_subsidy_tracker.plotting import ChartBuilder
from uk_subsidy_tracker.schemes import ro


def _prepare() -> pd.DataFrame:
    sm = pq.read_table(ro.DERIVED_DIR / "station_month.parquet").to_pandas()
    gb = sm[sm["country"] == "GB"]
    by_unit = (
        gb.groupby(["station_id"], sort=False, observed=True)
        .agg(payments=("ro_cost_gbp", "sum"))
        .reset_index()
    )
    by_unit = by_unit[by_unit["payments"] > 0].sort_values("payments", ascending=False).reset_index(drop=True)
    return by_unit


def main() -> None:
    by_unit = _prepare()
    if len(by_unit) == 0:
        return
    total = float(by_unit["payments"].sum())
    n = len(by_unit)
    cum_pct_payments = (by_unit["payments"].cumsum() / total * 100).tolist()
    cum_pct_projects = (np.arange(1, n + 1) / n * 100).tolist()
    total_bn = total / 1e9

    builder = ChartBuilder(title=f"RO cost concentration — £{total_bn:.1f}bn across {n} GB stations", height=700)
    fig = go.Figure()
    fig.add_trace(go.Scatter(x=[0, 100], y=[0, 100], mode="lines",
                              line=dict(color="#888", dash="dash"), name="Perfect equality"))
    fig.add_trace(go.Scatter(x=cum_pct_projects, y=cum_pct_payments, fill="tozeroy",
                              line=dict(color="rgb(214,39,40)"),
                              fillcolor="rgba(214,39,40,0.1)", name="RO payments Lorenz"))
    # 50% + 80% threshold markers
    for threshold in (50.0, 80.0):
        idx = int(next((i for i, v in enumerate(cum_pct_payments) if v >= threshold), n - 1))
        x = cum_pct_projects[idx]
        fig.add_annotation(x=x, y=threshold,
                           text=f"{idx+1} stations = {threshold:.0f}% of cost", showarrow=True)

    builder.save(fig, "subsidy_ro_concentration", export_twitter=True)


if __name__ == "__main__":
    main()
```

```python
"""RO forward projection — S5 drawdown to 2037 (flat scenario; no growth — RO is closed to new accreditations)."""
from __future__ import annotations
import pandas as pd
import plotly.graph_objects as go
import pyarrow.parquet as pq

from uk_subsidy_tracker.plotting import ChartBuilder
from uk_subsidy_tracker.schemes import ro


def _prepare() -> pd.DataFrame:
    fp = pq.read_table(ro.DERIVED_DIR / "forward_projection.parquet").to_pandas()
    return fp


def main() -> None:
    fp = _prepare()
    builder = ChartBuilder(title="RO forward-committed obligations to 2037", height=800)
    fig = builder.create_subplots(rows=2, cols=1, shared_xaxes=True,
                                   subplot_titles=["[1] Annual remaining committed (£bn/yr)",
                                                   "[2] Cumulative remaining bill"])

    # Group by year stacked by technology
    annual_total = fp.groupby("year", sort=True, observed=True).agg(
        cost_bn=("remaining_cost_gbp", lambda s: s.sum() / 1e9)
    ).reset_index()
    fig.add_trace(go.Bar(x=annual_total["year"], y=annual_total["cost_bn"]), row=1, col=1)

    annual_total["cum_bn"] = annual_total["cost_bn"].cumsum()
    fig.add_trace(go.Scatter(x=annual_total["year"], y=annual_total["cum_bn"],
                              mode="lines+markers", fill="tozeroy"), row=2, col=1)

    fig.add_vline(x=2037, line=dict(color="red", dash="dot"), annotation_text="2037 scheme close")

    builder.save(fig, "subsidy_ro_forward_projection", export_twitter=True)


if __name__ == "__main__":
    main()
```
