```python
"""Pydantic row models for RO Parquet grains.

Each row model's field declaration order IS the Parquet column order.
Changing the order requires a CHANGES.md "## Methodology versions" entry citing D-10.

Per CONTEXT:
  - D-09: country column on EVERY RO row model (GB / NI)
  - D-10: field-order discipline (Parquet column order = Pydantic declaration order)
  - D-11: mutualisation_gbp is nullable (non-null for OY 2021-22 only)
  - D-02/D-05: dual cost columns (ro_cost_gbp primary, ro_cost_gbp_eroc sensitivity,
              ro_cost_gbp_nocarbon sensitivity)

emit_schema_json is imported from schemas.cfd; it is scheme-agnostic and DRY-shared.
"""
from __future__ import annotations

from datetime import date
from pydantic import BaseModel, Field

# Import, do NOT re-declare — scheme-agnostic emitter shared via schemas.cfd (D-10 contract).
from uk_subsidy_tracker.schemas.cfd import emit_schema_json  # noqa: F401 (re-exported)


class RoStationMonthRow(BaseModel):
    """One row in ro/station_month.parquet.

    Field declaration order IS the Parquet column order (D-10).
    """
    station_id: str = Field(
        description="Ofgem station accreditation number",
        json_schema_extra={"dtype": "string"},
    )
    country: str = Field(
        description="'GB' or 'NI' per Ofgem register (D-09)",
        json_schema_extra={"dtype": "string"},
    )
    technology: str = Field(
        description="Ofgem technology_type",
        json_schema_extra={"dtype": "string"},
    )
    commissioning_window: str = Field(
        description="Banding window label (pre-2013, 2013/14, 2014/15, 2015/16, 2016/17)",
        json_schema_extra={"dtype": "string"},
    )
    month_end: date = Field(
        description="Month-end anchor of the obligation-month row",
        json_schema_extra={"dtype": "date", "unit": "ISO-8601"},
    )
    obligation_year: int = Field(
        description="Apr-Mar year containing output_period_end (D-08)",
        json_schema_extra={"dtype": "int64", "unit": "year"},
    )
    generation_mwh: float = Field(
        description="RO-eligible MWh",
        json_schema_extra={"dtype": "float64", "unit": "MWh"},
    )
    rocs_issued: float = Field(
        description="Ofgem-published ROCs issued (primary per D-01)",
        json_schema_extra={"dtype": "float64", "unit": "ROCs"},
    )
    rocs_computed: float = Field(
        description="generation_mwh × banding_factor_yaml (audit cross-check per D-01)",
        json_schema_extra={"dtype": "float64", "unit": "ROCs"},
    )
    banding_factor_yaml: float = Field(
        description="Lookup from ro_bandings.yaml (audit source)",
        json_schema_extra={"dtype": "float64", "unit": "ROCs/MWh"},
    )
    ro_cost_gbp: float = Field(
        description="Primary cost = rocs_issued × (buyout + recycle) + mutualisation_gbp (per D-02, D-11)",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    ro_cost_gbp_eroc: float = Field(
        description="Sensitivity: rocs_issued × e-ROC clearing (per D-02)",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    ro_cost_gbp_nocarbon: float = Field(
        description="Sensitivity: carbon-price=0 variant (per D-05)",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    gas_counterfactual_gbp: float = Field(
        description="compute_counterfactual() × generation_mwh",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    premium_gbp: float = Field(
        description="ro_cost_gbp - gas_counterfactual_gbp",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    mutualisation_gbp: float | None = Field(
        default=None,
        description="Mutualisation delta (null except OY 2021-22 per D-11)",
        json_schema_extra={"dtype": "float64", "unit": "GBP"},
    )
    methodology_version: str = Field(
        description="counterfactual.METHODOLOGY_VERSION stamp (D-12 chain)",
        json_schema_extra={"dtype": "string"},
    )


class RoAnnualSummaryRow(BaseModel):
    """One row in ro/annual_summary.parquet (per year × country per D-09)."""
    year: int = Field(json_schema_extra={"dtype": "int64", "unit": "year"})
    country: str = Field(json_schema_extra={"dtype": "string"})
    ro_generation_mwh: float = Field(json_schema_extra={"dtype": "float64", "unit": "MWh"})
    ro_cost_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    ro_cost_gbp_eroc: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    gas_counterfactual_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    premium_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    mutualisation_gbp: float | None = Field(default=None, json_schema_extra={"dtype": "float64", "unit": "GBP"})
    methodology_version: str = Field(json_schema_extra={"dtype": "string"})


class RoByTechnologyRow(BaseModel):
    """One row in ro/by_technology.parquet (per year × technology)."""
    year: int = Field(json_schema_extra={"dtype": "int64", "unit": "year"})
    technology: str = Field(json_schema_extra={"dtype": "string"})
    ro_generation_mwh: float = Field(json_schema_extra={"dtype": "float64", "unit": "MWh"})
    ro_cost_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    ro_cost_gbp_eroc: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    methodology_version: str = Field(json_schema_extra={"dtype": "string"})


class RoByAllocationRoundRow(BaseModel):
    """One row in ro/by_allocation_round.parquet.

    RO has no 'allocation round' axis (unlike CfD); use commissioning_window per RESEARCH §5.
    """
    year: int = Field(json_schema_extra={"dtype": "int64", "unit": "year"})
    commissioning_window: str = Field(json_schema_extra={"dtype": "string"})
    ro_generation_mwh: float = Field(json_schema_extra={"dtype": "float64", "unit": "MWh"})
    ro_cost_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    ro_cost_gbp_eroc: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    methodology_version: str = Field(json_schema_extra={"dtype": "string"})


class RoForwardProjectionRow(BaseModel):
    """One row in ro/forward_projection.parquet (projected year × technology)."""
    year: int = Field(json_schema_extra={"dtype": "int64", "unit": "year"})
    technology: str = Field(json_schema_extra={"dtype": "string"})
    remaining_committed_mwh: float = Field(json_schema_extra={"dtype": "float64", "unit": "MWh"})
    remaining_cost_gbp: float = Field(json_schema_extra={"dtype": "float64", "unit": "GBP"})
    station_count_active: int = Field(json_schema_extra={"dtype": "int64", "unit": "count"})
    avg_banding_factor: float = Field(json_schema_extra={"dtype": "float64", "unit": "ROCs/MWh"})
    methodology_version: str = Field(json_schema_extra={"dtype": "string"})
```
