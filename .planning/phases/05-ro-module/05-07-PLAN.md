```python
import importlib
from contextlib import ExitStack
from pathlib import Path
from unittest.mock import MagicMock, patch

import pytest


def _setup_ro_raw_tree(tmp_path: Path) -> None:
    """Create fake data/raw/ofgem/* files with predictable content."""
    ofgem = tmp_path / "raw" / "ofgem"
    ofgem.mkdir(parents=True, exist_ok=True)
    (ofgem / "ro-register.xlsx").write_bytes(b"FAKE_REGISTER_XLSX\n")
    (ofgem / "ro-generation.csv").write_text("station_id,output_period_end,generation_mwh,rocs_issued\n")
    (ofgem / "roc-prices.csv").write_text("obligation_year,buyout_gbp_per_roc,recycle_gbp_per_roc,eroc_clearing_gbp_per_roc,mutualisation_gbp_per_roc\n")


def test_ro_refresh_converges_on_unchanged_upstream(tmp_path, monkeypatch):
    """After RO refresh(), upstream_changed() returns False (Phase 4 D-18 per-scheme invariant)."""
    ro_refresh = importlib.import_module("uk_subsidy_tracker.schemes.ro._refresh")
    monkeypatch.setattr(ro_refresh, "DATA_DIR", tmp_path)

    _setup_ro_raw_tree(tmp_path)

    with ExitStack() as stack:
        stack.enter_context(patch(
            "uk_subsidy_tracker.data.ofgem_ro.download_ofgem_ro_register",
            side_effect=lambda: _setup_ro_raw_tree(tmp_path) or (tmp_path / "raw" / "ofgem" / "ro-register.xlsx"),
        ))
        stack.enter_context(patch(
            "uk_subsidy_tracker.data.ofgem_ro.download_ofgem_ro_generation",
            side_effect=lambda: _setup_ro_raw_tree(tmp_path) or (tmp_path / "raw" / "ofgem" / "ro-generation.csv"),
        ))
        stack.enter_context(patch(
            "uk_subsidy_tracker.data.roc_prices.download_roc_prices",
            side_effect=lambda: _setup_ro_raw_tree(tmp_path) or (tmp_path / "raw" / "ofgem" / "roc-prices.csv"),
        ))

        ro_refresh.refresh()

    # After refresh, upstream_changed must return False
    assert ro_refresh.upstream_changed() is False, "refresh-loop invariant broken for RO"


def test_manifest_includes_both_schemes_after_end_to_end_refresh(tmp_path, monkeypatch):
    """Plan 05-07: multi-scheme refresh produces manifest.json with cfd.* AND ro.* entries."""
    from uk_subsidy_tracker.publish import manifest as manifest_mod

    # Setup: tmp derived_root with both cfd/ and ro/ subdirs populated
    derived_root = tmp_path / "derived"
    for scheme in ("cfd", "ro"):
        (derived_root / scheme).mkdir(parents=True)
        for grain in ("station_month", "annual_summary", "by_technology",
                      "by_allocation_round", "forward_projection"):
            import pyarrow as pa
            import pyarrow.parquet as pq
            tbl = pa.table({"methodology_version": ["0.1.0"]})
            pq.write_table(tbl, derived_root / scheme / f"{grain}.parquet")

    raw_dir = tmp_path / "raw"
    raw_dir.mkdir()

    from uk_subsidy_tracker.schemes import cfd, ro
    manifest = manifest_mod.build(
        version="v2026.04",
        schemes=[("cfd", cfd), ("ro", ro)],
        derived_root=derived_root,
        raw_dir=raw_dir,
        output_path=tmp_path / "manifest.json",
    )

    ids = set(d.id for d in manifest.datasets)
    cfd_ids = {i for i in ids if i.startswith("cfd.")}
    ro_ids = {i for i in ids if i.startswith("ro.")}
    assert len(cfd_ids) == 5, f"CfD: got {cfd_ids}"
    assert len(ro_ids) == 5, f"RO: got {ro_ids}"
```
