```python
REF_TOLERANCE_PCT: float = 3.0
"""REF Constable 2025 primary RO benchmark tolerance (D-14 HARD BLOCK).

Per CONTEXT D-14: if divergence exceeds 3%, investigate root cause BEFORE raising tolerance.
Investigate in this order:
  1. REF scope difference: is REF excluding NIRO? including mutualisation? SY vs CY?
     Document scope-delta in benchmarks.yaml audit header.
  2. Banding error: R1 regression in ro_bandings.yaml or ofgem_ro.load_*?
  3. Carbon-price extension regression: D-05 2005-2017 values wrong?
Only raise tolerance with CHANGES.md `## Methodology versions` entry (D-07)."""

_TOLERANCE_BY_SOURCE: dict[str, float] = {
    "ofgem_transparency": OFGEM_TOLERANCE_PCT,
    "obr_efo": OBR_EFO_TOLERANCE_PCT,
    "desnz_energy_trends": DESNZ_TOLERANCE_PCT,
    "hoc_library": HOC_LIBRARY_TOLERANCE_PCT,
    "nao_audit": NAO_TOLERANCE_PCT,
    "ref_constable": REF_TOLERANCE_PCT,  # Plan 05-09
}


@pytest.fixture(scope="module")
def ro_annual_totals_gbp_bn() -> dict[int, float]:
    """Pipeline yearly RO totals in £bn (GB-only per D-12), keyed by calendar year.

    Sums ro_cost_gbp from data/derived/ro/station_month.parquet filtered to country='GB'.
    Requires Plan 05-05 + 05-07 to have produced derived Parquet; tests assume a build
    has run (refresh_all or direct ro.rebuild_derived call in CI before pytest).
    """
    import pyarrow.parquet as pq
    from uk_subsidy_tracker.schemes import ro

    path = ro.DERIVED_DIR / "station_month.parquet"
    if not path.exists():
        pytest.fail(
            f"{path} missing — REF reconciliation requires RO derived Parquet. "
            "Run `uv run python -c \"from uk_subsidy_tracker.schemes import ro; "
            "ro.rebuild_derived()\"` before pytest."
        )

    df = pq.read_table(path).to_pandas()
    gb = df[df["country"] == "GB"].copy()
    gb["cy"] = gb["month_end"].dt.year
    totals_gbp = gb.groupby("cy")["ro_cost_gbp"].sum()
    return (totals_gbp / 1e9).astype(float).to_dict()


def _ref_entries():
    """Load REF Constable entries; fail loud if empty (D-14 no-skip policy)."""
    from tests.fixtures import load_benchmarks
    b = load_benchmarks()
    entries = b.ref_constable
    if not entries:
        pytest.fail(
            "benchmarks.yaml::ref_constable is empty — Phase 5 Plan 05-09 failed to "
            "transcribe REF Constable 2025 Table 1. Fix Plan 05-09 Task 1 before proceeding."
        )
    return entries


@pytest.mark.parametrize(
    "entry",
    _ref_entries(),
    ids=lambda e: f"ref_constable-{e.year}",
)
def test_ref_constable_ro_reconciliation(entry, ro_annual_totals_gbp_bn):
    """RO-06 / D-14 / ROADMAP SC #3: pipeline ro_cost_gbp aggregate ±3% of REF per-year.

    HARD BLOCK: if this fails, investigate BEFORE raising tolerance.
    """
    ours = ro_annual_totals_gbp_bn.get(entry.year)
    if ours is None:
        pytest.fail(
            f"Pipeline has no data for year {entry.year} — either the RO derived Parquet "
            f"was not built before this test or station_month.parquet filtering to "
            f"country='GB' dropped the year. Check that Plan 05-05 built annual_summary for {entry.year}."
        )

    divergence_pct = abs(ours - entry.value_gbp_bn) / entry.value_gbp_bn * 100.0
    assert divergence_pct <= REF_TOLERANCE_PCT, (
        f"REF Constable reconciliation FAILED for {entry.year} (D-14 HARD BLOCK):\n"
        f"  pipeline:    £{ours:.4f} bn\n"
        f"  REF Table 1: £{entry.value_gbp_bn:.4f} bn\n"
        f"  divergence:  {divergence_pct:.2f}% (> {REF_TOLERANCE_PCT}% tolerance)\n"
        f"\nInvestigate before adjusting tolerance (per D-14):\n"
        f"  1. REF scope — is it excluding NIRO? including mutualisation? SY vs CY?\n"
        f"     (Check benchmarks.yaml audit header vs our D-12 scope.)\n"
        f"  2. Banding error — cross-check station_month rocs_computed vs rocs_issued.\n"
        f"  3. Carbon-price extension — is DEFAULT_CARBON_PRICES[{entry.year}] sensible?\n"
        f"URL: {entry.url}"
    )
```
