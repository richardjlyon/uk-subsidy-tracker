def validate() -> list[str]:
    """Post-rebuild sanity checks (D-04). Returns warnings (empty = clean)."""
    warnings: list[str] = []
    import pyarrow.parquet as pq
    import pandas as pd

    # Check 1 — banding divergence (D-04 Check 1)
    sm_path = DERIVED_DIR / "station_month.parquet"
    if sm_path.exists():
        sm = pq.read_table(sm_path).to_pandas()
        if len(sm) > 0 and "rocs_issued" in sm.columns and "rocs_computed" in sm.columns:
            nonzero_issued = sm[sm["rocs_issued"] > 0].copy()
            nonzero_issued["delta_pct"] = (
                (nonzero_issued["rocs_issued"] - nonzero_issued["rocs_computed"]).abs()
                / nonzero_issued["rocs_issued"]
            )
            divergent = nonzero_issued[nonzero_issued["delta_pct"] > 0.01]
            station_count = nonzero_issued["station_id"].nunique() if len(nonzero_issued) else 0
            divergent_stations = divergent["station_id"].nunique() if len(divergent) else 0
            if divergent_stations > 10 or (station_count > 0 and divergent_stations / station_count > 0.05):
                warnings.append(
                    f"D-04 Check 1 (banding divergence): {divergent_stations} stations "
                    f"have >1% Ofgem-vs-YAML rocs delta (threshold: >10 or >5% of {station_count})"
                )

    # Check 2 — REF benchmark drift (D-04 Check 2); cheap post-rebuild drift warner.
    # This is the cheap runtime equivalent of the hard-block REF reconciliation in
    # tests/test_benchmarks.py (D-14). validate() warns at >3% drift; test_benchmarks
    # fails CI at the same threshold. Complementary layers, not redundant.
    annual_path = DERIVED_DIR / "annual_summary.parquet"
    if annual_path.exists():
        ann = pq.read_table(annual_path).to_pandas()
        if "ro_cost_gbp" in ann.columns:
            # Cheap negative-sanity short-circuit
            if (ann["ro_cost_gbp"] < 0).any():
                warnings.append("D-04 Check 2: negative ro_cost_gbp in annual_summary")

            # REF drift aggregate check (ref_constable benchmark; Plan 05-09 fixture)
            # Short-circuits silently if the fixture isn't loadable at validate()-time
            # (e.g. benchmarks.yaml absent during partial rebuild) — test_benchmarks
            # remains the hard-block.
            try:
                import yaml  # pyyaml; already a project dep
                ref_path = PROJECT_ROOT / "src" / "uk_subsidy_tracker" / "data" / "benchmarks.yaml"
                if ref_path.exists() and "year" in ann.columns:
                    raw = yaml.safe_load(ref_path.read_text())
                    ref_block = (raw or {}).get("ref_constable", {})
                    # Expect ref_block["annual_gbp"] as {year:int -> gbp:float} for 2011-2022
                    ref_annual = ref_block.get("annual_gbp") or {}
                    if ref_annual:
                        window = ann[(ann["year"] >= 2011) & (ann["year"] <= 2022)]
                        # Cost across all countries for the window
                        built = float(window["ro_cost_gbp"].sum())
                        ref_total = float(sum(
                            float(v) for k, v in ref_annual.items()
                            if 2011 <= int(k) <= 2022
                        ))
                        if ref_total > 0:
                            drift_pct = abs(built - ref_total) / ref_total * 100.0
                            if drift_pct > 3.0:
                                warnings.append(
                                    f"D-04 Check 2 (REF drift): RO 2011-2022 built aggregate "
                                    f"£{built:,.0f} vs ref_constable £{ref_total:,.0f} "
                                    f"(drift {drift_pct:.2f}% > 3.0% tolerance)"
                                )
            except Exception:
                # validate() must never raise; drift detection is best-effort here.
                # Hard-block reconciliation lives in tests/test_benchmarks.py.
                pass

    # Check 3 — methodology_version consistency (D-04 Check 3)
    for grain in ["station_month", "annual_summary", "by_technology",
                  "by_allocation_round", "forward_projection"]:
        path = DERIVED_DIR / f"{grain}.parquet"
        if path.exists():
            df = pq.read_table(path).to_pandas()
            if "methodology_version" in df.columns:
                unique = set(df["methodology_version"].dropna().unique())
                if unique != {METHODOLOGY_VERSION}:
                    warnings.append(
                        f"D-04 Check 3: {grain}.parquet methodology_version drift — "
                        f"expected {{{METHODOLOGY_VERSION!r}}}, got {unique}"
                    )

    # Check 4 — forward-projection sanity (D-04 Check 4)
    fp_path = DERIVED_DIR / "forward_projection.parquet"
    if fp_path.exists():
        fp = pq.read_table(fp_path).to_pandas()
        if (fp["remaining_cost_gbp"] < 0).any():
            warnings.append("D-04 Check 4: negative remaining_cost_gbp in forward_projection")
        if "remaining_committed_mwh" in fp.columns and "year" in fp.columns and "technology" in fp.columns:
            for tech, sub in fp.sort_values("year").groupby("technology"):
                mwh = sub["remaining_committed_mwh"].to_numpy()
                if len(mwh) >= 2:
                    import numpy as np
                    with np.errstate(divide="ignore", invalid="ignore"):
                        jumps = np.abs(np.diff(mwh)) / np.where(mwh[:-1] > 0, mwh[:-1], 1)
                    if (jumps > 0.5).any():
                        warnings.append(
                            f"D-04 Check 4: forward_projection remaining_committed_mwh jumps >50% "
                            f"across adjacent years for technology={tech!r}"
                        )

    return warnings

def _sha256(path: Path) -> str:
    h = hashlib.sha256()
    with path.open("rb") as f:
        for chunk in iter(lambda: f.read(1 << 16), b""):
            h.update(chunk)
    return h.hexdigest()


def upstream_changed() -> bool:
    for rel in _RAW_FILES:
        raw = DATA_DIR / rel
        meta = raw.with_suffix(raw.suffix + ".meta.json")
        if not raw.exists() or not meta.exists():
            return True
        sidecar = json.loads(meta.read_text())
        if _sha256(raw) != sidecar.get("sha256"):
            return True
    return False

def _obligation_year_label(month_end: pd.Timestamp) -> str:
    """Apr-Mar span containing month_end, formatted 'YYYY-YY' (e.g. 2021-22).

    March month_end → previous OY (ends Mar 31). April month_end → new OY.
    """
    ts = pd.Timestamp(month_end)

        start_year = ts.year
    else:
        start_year = ts.year - 1
    return f"{start_year}-{str(start_year + 1)[-2:]}"


def _commissioning_window_label(commissioning_date: pd.Timestamp | None) -> str:
    """REF Table 1 window label. None / pre-2013 → 'pre-2013'; else 'YYYY/YY'."""
    if commissioning_date is None or pd.isna(commissioning_date):
        return "pre-2013"
    ts = pd.Timestamp(commissioning_date)
    if ts < pd.Timestamp("2013-04-01"):
        return "pre-2013"
    start_year = ts.year if ts.month >= 4 else ts.year - 1
    return f"{start_year}/{str(start_year + 1)[-2:]}"

def build_station_month(output_dir: Path) -> pd.DataFrame:
    """Raw Ofgem + ROC prices + bandings → data/derived/ro/station_month.parquet."""
    # (1) Load raw — pandera validation happens inside each loader
    gen = load_ofgem_ro_generation()
    reg = load_ofgem_ro_register()
    prices = load_roc_prices()
    bandings = load_ro_bandings()

    # (2) Merge generation (monthly) with register (per-station metadata)
    df = gen.merge(
        reg[["station_id", "technology_type", "country", "commissioning_date"]],
        on="station_id", how="left",
    )
    df = df.rename(columns={"technology_type": "technology"})

    # (3) Month-end anchor (snap to month end) + obligation_year label (D-08)
    df["month_end"] = pd.to_datetime(df["output_period_end"]) + pd.offsets.MonthEnd(0)
    df["obligation_year"] = df["month_end"].apply(_obligation_year_label)

    # (4) Normalise country to GB/NI per D-09
    country_map = {
        "England": "GB", "Scotland": "GB", "Wales": "GB", "GB": "GB",
        "Northern Ireland": "NI", "NI": "NI",
    }
    df["country"] = df["country"].map(country_map).fillna("GB")

    # (5) Commissioning-window label (for downstream by_allocation_round rollup)
    df["commissioning_window"] = df["commissioning_date"].apply(_commissioning_window_label)

    # (6) Banding lookup (D-01 audit cross-check)
    def _lookup_banding(row):
        try:
            return bandings.lookup(
                technology=row["technology"],
                country=row["country"],
                commissioning_date=row["commissioning_date"].date() if pd.notna(row["commissioning_date"]) else pd.Timestamp("2005-01-01").date(),
            )
        except KeyError:
            return float("nan")

    df["banding_factor_yaml"] = df.apply(_lookup_banding, axis=1)
    df["rocs_computed"] = df["generation_mwh"] * df["banding_factor_yaml"]

    # (7) Join roc-prices on obligation_year (buyout + recycle + e-ROC + mutualisation)
    df = df.merge(prices, on="obligation_year", how="left")

    # (8) Compute cost columns (D-02, D-05, D-11)
    mutualisation_per_roc = df["mutualisation_gbp_per_roc"].fillna(0.0)
    df["ro_cost_gbp"] = df["rocs_issued"] * (
        df["buyout_gbp_per_roc"] + df["recycle_gbp_per_roc"] + mutualisation_per_roc
    )
    df["ro_cost_gbp_eroc"] = df["rocs_issued"] * df["eroc_clearing_gbp_per_roc"].fillna(
        df["buyout_gbp_per_roc"] + df["recycle_gbp_per_roc"]
    )

    # no-carbon sensitivity: same (buyout+recycle) since ROC price is carbon-independent;
    # this column is reserved for future sensitivity where carbon is re-derived per D-05.
    # In Phase 5 we set it equal to ro_cost_gbp minus carbon-attributable counterfactual delta.
    # Simplest behavioural equivalent: same as ro_cost_gbp (carbon does not enter ROC pricing).
    df["ro_cost_gbp_nocarbon"] = df["ro_cost_gbp"]

    # (9) Gas counterfactual via compute_counterfactual() at row's calendar year
    df["_cy_year"] = df["month_end"].dt.year.astype("int64")

    def _row_counterfactual(year: int) -> float:
        # compute_counterfactual signature reused from CfD path
        result = compute_counterfactual(
            year=int(year),
            carbon_prices=DEFAULT_CARBON_PRICES,
        )
        # compute_counterfactual returns £/MWh; downstream multiplies by generation
        if isinstance(result, dict):
            return float(result.get("total_gbp_per_mwh", result.get("total", 0.0)))
        return float(result)

    # Use vectorised apply — unique years only for efficiency (~20 years × 500k rows)
    unique_years = df["_cy_year"].unique()
    year_cf = {y: _row_counterfactual(int(y)) for y in unique_years}
    df["gas_counterfactual_gbp"] = df["_cy_year"].map(year_cf) * df["generation_mwh"]
    df["premium_gbp"] = df["ro_cost_gbp"] - df["gas_counterfactual_gbp"]

    # Mutualisation_gbp per row (null except OY 2021-22 per D-11)
    df["mutualisation_gbp"] = df["rocs_issued"] * df["mutualisation_gbp_per_roc"]

    # (10) D-12 methodology_version column
    df["methodology_version"] = METHODOLOGY_VERSION

    # (11) Sort + column-order + validate + write
    columns = list(RoStationMonthRow.model_fields.keys())
    df = df[columns].sort_values(["station_id", "month_end"], kind="mergesort").reset_index(drop=True)
    df = station_month_schema.validate(df)

    output_dir.mkdir(parents=True, exist_ok=True)
    _write_parquet(df, output_dir / "station_month.parquet")
    emit_schema_json(RoStationMonthRow, output_dir / "station_month.schema.json")
    return df

"""RO forward projection — per-station accreditation-end anchor (RESEARCH §6 Option c).

D-21: deterministic current-year anchor from max(output_period_end).year — NEVER
pd.Timestamp.now().year (breaks test_determinism).

Per RESEARCH §6:
  accreditation_end_year = accreditation_date.year + 20  (or expected_end_date if present)
  avg_annual_generation = historical_sum / n_years_metered
  remaining_years = max(0, min(accreditation_end_year, 2037) - current_year_anchor)
  remaining_committed_mwh = avg_annual_generation × remaining_years
  remaining_cost_gbp = remaining_committed_mwh × avg_banding_factor × (buyout + recycle)[latest_oy]

RO scheme closes 2037-03-31 for 20-year accreditations starting 2017-03-31.
2037 is the hard cap.
"""
from __future__ import annotations
from pathlib import Path

import pandas as pd
import pyarrow.parquet as pq

from uk_subsidy_tracker.counterfactual import METHODOLOGY_VERSION
from uk_subsidy_tracker.data.ofgem_ro import load_ofgem_ro_register, load_ofgem_ro_generation
from uk_subsidy_tracker.data.roc_prices import load_roc_prices
from uk_subsidy_tracker.schemas.ro import RoForwardProjectionRow, emit_schema_json
from uk_subsidy_tracker.schemes.cfd.cost_model import _write_parquet

_RO_SCHEME_CLOSE_YEAR = 2037


def build_forward_projection(output_dir: Path) -> pd.DataFrame:
    """data/derived/ro/forward_projection.parquet — 2026→2037 drawdown by technology."""
    gen = load_ofgem_ro_generation()
    reg = load_ofgem_ro_register()
    prices = load_roc_prices()

    # Deterministic anchor — D-21 Phase 4 contract
    current_year_anchor = int(pd.to_datetime(gen["output_period_end"]).max().year)

    # Per-station accreditation end
    reg = reg.copy()
    reg["accreditation_end_year"] = reg.apply(
        lambda r: (
            r["expected_end_date"].year if pd.notna(r.get("expected_end_date"))
            else (r["accreditation_date"].year + 20 if pd.notna(r.get("accreditation_date")) else _RO_SCHEME_CLOSE_YEAR)
        ),
        axis=1,
    ).astype("int64")
    reg["accreditation_end_year"] = reg["accreditation_end_year"].clip(upper=_RO_SCHEME_CLOSE_YEAR)

    # Historical per-station annual generation average
    station_annual = (
        gen.assign(year=pd.to_datetime(gen["output_period_end"]).dt.year)
        .groupby(["station_id", "year"], observed=True)["generation_mwh"].sum()
        .reset_index()
        .groupby("station_id", observed=True)["generation_mwh"].mean()
        .rename("avg_annual_mwh")
        .reset_index()
    )

    stations = reg.merge(station_annual, on="station_id", how="left")
    stations["avg_annual_mwh"] = stations["avg_annual_mwh"].fillna(0.0)
    stations = stations.rename(columns={"technology_type": "technology"})

    # Most-recent price (latest OY in roc-prices.csv)
    latest_prices = prices.sort_values("obligation_year").tail(1).iloc[0]
    latest_price_per_roc = float(latest_prices["buyout_gbp_per_roc"] + latest_prices["recycle_gbp_per_roc"])

    # For each projected year, count active stations + compute remaining MWh/cost
    rows = []
    for y in range(current_year_anchor + 1, _RO_SCHEME_CLOSE_YEAR + 1):
        active = stations[stations["accreditation_end_year"] >= y].copy()
        if active.empty:
            continue
        # banding_factor proxy — use 1.0 as fallback; real banding lookup is in station_month
        # Approx via average band by technology
        for tech, sub in active.groupby("technology", observed=True):
            mwh = float(sub["avg_annual_mwh"].sum())
            station_count = int(len(sub))
            # assume avg banding factor = 1.0 proxy (RO aggregate cost is via station_month path;
            # this is a forward-looking "what's committed" projection)
            avg_bf = 1.0
            rows.append(
                {
                    "year": int(y),
                    "technology": tech,
                    "remaining_committed_mwh": mwh,
                    "remaining_cost_gbp": mwh * avg_bf * latest_price_per_roc,
                    "station_count_active": station_count,
                    "avg_banding_factor": avg_bf,
                    "methodology_version": METHODOLOGY_VERSION,
                }
            )

    df = pd.DataFrame(rows)
    if df.empty:
        # Edge case: no future years — write empty Parquet with correct schema
        df = pd.DataFrame(
            {col: pd.Series(dtype=dtype) for col, dtype in [
                ("year", "int64"), ("technology", "string"),
                ("remaining_committed_mwh", "float64"), ("remaining_cost_gbp", "float64"),
                ("station_count_active", "int64"), ("avg_banding_factor", "float64"),
                ("methodology_version", "string"),
            ]}
        )

    columns = list(RoForwardProjectionRow.model_fields.keys())
    df = df[columns].sort_values(["year", "technology"], kind="mergesort").reset_index(drop=True)
    _write_parquet(df, output_dir / "forward_projection.parquet")
    emit_schema_json(RoForwardProjectionRow, output_dir / "forward_projection.schema.json")
    return df
