    ```python
    def _unified_annual_frame() -> pd.DataFrame:
        """Merge the XLSX monthly aggregate (pre-2019) with the transcribed
        SY17-SY23 annual aggregate into a single long-form frame keyed on
        (year, country, technology).
        """
        monthly = parse_xlsx_to_monthly()
        pre2019 = (
            monthly[monthly["year"] < 2018]
            .groupby(["year", "country", "technology"], sort=True, as_index=False)
            .agg({"generation_mwh": "sum", "rocs_issued": "sum"})
        )
        pre2019["scheme_year"] = pre2019["year"].apply(lambda y: f"SY{str(int(y)-2001)[-2:].zfill(2)}")
        pre2019["ro_cost_gbp_nominal"] = None
        pre2019["source_pdf_url"] = None

        annual = load_annual_aggregate_csv().copy()
        annual["generation_mwh"] = annual["generation_gwh"].astype(float) * 1000.0
        annual = annual[["scheme_year", "year", "country", "technology", "generation_mwh", "rocs_issued", "ro_cost_gbp_nominal", "source_pdf_url"]]

        unified = pd.concat([pre2019, annual], ignore_index=True)
        unified = unified.drop_duplicates(subset=["year", "country", "technology"], keep="last")
        return unified.sort_values(["year", "country", "technology"], kind="mergesort").reset_index(drop=True)
    ```

    def build_annual_summary_aggregate(output_dir: Path) -> pd.DataFrame:
        """Build annual_summary.parquet from Ofgem aggregates (D-04 nullable eroc).

        Pure function of data/raw/ofgem/ + data/raw/ons/ content (D-21) — no
        clock, no randomness. Column order = RoAnnualSummaryRow field
        declaration order (D-10). Final stable sort: (year, country).
        """
        output_dir.mkdir(parents=True, exist_ok=True)

        unified = _unified_annual_frame()
        prices = load_roc_prices_csv().copy()
        prices["year"] = prices["obligation_year"].apply(_obligation_year_to_end_year)

        by_yc = (
            unified.groupby(["year", "country"], sort=True, as_index=False)
            .agg({"generation_mwh": "sum", "rocs_issued": "sum", "ro_cost_gbp_nominal": "sum"})
        )
        by_yc = by_yc.merge(
            prices[["year", "buyout_gbp_per_roc", "mutualisation_gbp_total"]],
            on="year", how="left",
        )

        by_yc["ro_cost_gbp"] = by_yc["ro_cost_gbp_nominal"].where(
            by_yc["ro_cost_gbp_nominal"].notna(),
            by_yc["rocs_issued"] * by_yc["buyout_gbp_per_roc"],
        )
        by_yc["mutualisation_gbp"] = by_yc["mutualisation_gbp_total"]

        cf = _annual_counterfactual_gbp_per_mwh()
        by_yc["gas_counterfactual_gbp"] = by_yc.apply(
            lambda r: (cf.get(int(r["year"]), 0.0) * r["generation_mwh"]) if r["generation_mwh"] else 0.0,
